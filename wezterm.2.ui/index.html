WezTerm Configuration 2: UI

December 25, 2024 · 5 min read

In order to configure the WezTerm to have a nicer style, create the file ui.lua and update your wezterm.lua file like this:

1-- ... the previous Configuration
2local ui = require("ui")
3
4ui.set_ui(config)
5ui.set_smart_splits(config)

And in the ui.lua file add:

  1local wezterm = require("wezterm")
  2
  3---@class WezModule
  4local M = {}
  5
  6---@type number|nil
  7M.last_tab = nil
  8
  9---@param tab_info table
 10---@return string
 11local function get_tab_title(tab_info)
 12 local title = tab_info.tab_title
 13 return (title and #title > 0) and title or tab_info.active_pane.title
 14end

 16---@param window table
 17local function create_workspace_status(window)
 18 local status = {
 19  background = { Background = { Color = "rgba(35, 33, 54, 0.95)" } },
 20  arrow = { Foreground = { Color = "rgba(35, 33, 54, 0.95)" } },
 21 }
 22
 23 if window:leader_is_active() then
 24  status.background = { Background = { Color = "#eb6f92" } }
 25 end
 26
 27 if window:active_tab():tab_id() ~= 0 then
 28  status.arrow = { Foreground = { Color = "#1e2030" } }
 29 end
 30
 31 return status
 32end

 34---@param battery_state string
 35---@return string
 36local function get_battery_icon(battery_state)
 37 local icons = {
 38  ["AC attached"] = "🔌",
 39  ["Full"] = "🔌",
 40  ["Unknown"] = "🔌",
 41  ["Finishing charge"] = "🔌",
 42  ["Charging"] = "🔌",
 43  ["Discharging"] = "🔋",

 49---@param time_seconds number
 50---@return string
 51local function format_time_remaining(time_seconds)
 52 if time_seconds <= 0 then
 53  return ""
 54 end
 55 local hours = math.floor(time_seconds / 3600)
 56 local minutes = math.floor((time_seconds % 3600) / 60)
 57 return string.format(" ⏳%02d:%02d", hours, minutes)
 58end
 59
 60---@param battery table
 61---@return string
 62local function format_battery_info(battery)
 63 local icon = get_battery_icon(battery.state)
 64 local percentage = string.format("%.0f%%", battery.state_of_charge * 100)
 65 local time_remaining = 0
 66
 67 if battery.state == "Discharging" then
 68  time_remaining = battery.time_to_empty or 0
 69 elseif battery.state == "Charging" then
 70  time_remaining = battery.time_to_full or 0
 71 end
 72
 73 return string.format(" %s%s%s", icon, percentage, format_time_remaining(time_remaining))
 74end

 76---@class StyleColors
 77---@field edge string
 78---@field background string
 79---@field foreground string
 80
 81---@param text string
 82---@param colors StyleColors
 83---@return table[]
 84local function create_styled_section(text, colors)
 85 return {
 86  { Background = { Color = colors.edge } },
 87  { Foreground = { Color = colors.background } },
 88  { Text = "" },
 89  { Background = { Color = colors.background } },
 90  { Foreground = { Color = colors.foreground } },
 91  { Text = text },
 92  { Background = { Color = colors.edge } },
 93  { Foreground = { Color = colors.background } },
 94  { Text = "" },
 95 }
 96end

 98---@param window table
 99local function update_right_status(window)
100 local colors = {
101  edge = "rgba(35, 33, 54, 0.95)",
102  background = "#9ccfd8",
103  foreground = "#2e3440",
104 }
105
106 local date = wezterm.strftime("📅 %b, %d ")
107 local time = wezterm.strftime("🕓 %H:%M ")
108 local battery = ""
109
110 for _, b in ipairs(wezterm.battery_info()) do
111  battery = battery .. format_battery_info(b)
112 end
113
114 local components = { { Text = "" } }
115 local sections = { date, time, battery }
116
117 for i, section in ipairs(sections) do
118  for _, item in ipairs(create_styled_section(section, colors)) do
119   table.insert(components, item)
120  end
121
122  if i < #sections then
123   table.insert(components, { Text = " " })
124   table.insert(components, { Text = "" })
125  end
126 end
127
128 window:set_right_status(wezterm.format(components))
129end

131---@param tab table
132---@param max_width number
133---@return table
134local function format_tab(tab, max_width)
135 local colors = {
136  edge = "rgba(35, 33, 54, 0.95)",
137  background = tab.is_active and "#ebbcba" or "#232136",
138  foreground = tab.is_active and "#2e3440" or "#d8dee9",
139 }
140
141 local icon = tab.is_active and "💻" or "🖥️"
142 local title = wezterm.truncate_right(get_tab_title(tab), max_width - 2)
143
144 return create_styled_section(icon .. " " .. title, colors)
145end
146---@param config table
147function M.set_ui(config)
148 wezterm.on("update-status", function(window, _)
149  local status = create_workspace_status(window)
150  local workspace_text = "   " .. window:active_workspace() .. " "
151
152  window:set_left_status(wezterm.format({
153   status.background,
154   { Text = workspace_text },
155   status.arrow,
156   { Text = "" },
157  }))
158
159  update_right_status(window)
160 end)
161
162 wezterm.on("format-tab-title", function(tab, _, _, _, _, max_width)
163  return format_tab(tab, max_width)
164 end)
165
166 return config
167end

182---@param resize_or_move "resize"|"move"
183---@param key string
184---@return table
185local function create_split_nav(resize_or_move, key)
186 local mods = resize_or_move == "resize" and "META" or "CTRL"
187
188 return {
189  key = key,
190  mods = mods,
191  action = wezterm.action_callback(function(win, pane)
192   if is_vim(pane) then
193    win:perform_action({
194     SendKey = { key = key, mods = mods },
195    }, pane)
196   else
197    local action = resize_or_move == "resize" and { AdjustPaneSize = { get_direction(key), 3 } }
198     or { ActivatePaneDirection = get_direction(key) }
199    win:perform_action(action, pane)
200   end
201  end),
202 }
203end
204
205---@param config table
206function M.set_smart_splits(config)
207 local keys = { "h", "j", "k", "l" }
208 for _, key in ipairs(keys) do
209  table.insert(config.keys, create_split_nav("move", key))
210  table.insert(config.keys, create_split_nav("resize", key))
211 end
212end
213
214return M
